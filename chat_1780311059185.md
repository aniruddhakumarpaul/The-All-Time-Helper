# 1. Generate 1000 Mock Product Descriptions
def generate_product_catalog():
    categories = ["Electronics", "Apparel", "Home & Furniture", "Books & Literature"]
    colors = ["red", "blue", "green", "black", "white", "gold"]
    adjectives = ["premium", "lightweight", "modern", "durable", "classic", "breathable"]
    
    catalog = []
    for i in range(1, 1001):
        cat = categories[i % len(categories)]
        color = colors[i % len(colors)]
        adj = adjectives[i % len(adjectives)]
        
        # Adding variety to descriptions
        if cat == "Electronics":
            desc = f"A {adj} {color} smart device with wireless connectivity and high battery capacity. Model {i}."
        elif cat == "Apparel":
            desc = f"A {adj} {color} cotton jacket designed with custom stitching and slim fit design. Style {i}."
        elif cat == "Home & Furniture":
            desc = f"A {adj} {color} oak table with polished surface, suitable for modern dining rooms. Item {i}."
        else: # Books
            desc = f"A {adj} {color} hardback novel detailing space travel adventures, authored by model {i}."
            
        catalog.append({"id": i, "description": desc, "category": cat, "color": color})
    return catalog

def run_indexing():
    print("\n" + "="*60)
    print("      VECTOR DATABASE & FAISS MULTILINGUAL INDEXING")
    print("="*60)
    
    catalog = generate_product_catalog()
    print(f"[+] Successfully generated {len(catalog)} product descriptions.")
    print(f" -> Example: {catalog[0]['description']}")
    
    # Multilingual Queries to test
    test_queries = [
        "chaqueta de algodón roja" , # Spanish for "red cotton jacket" -> Should match Apparel (red)
        "appareil intelligent noir", # French for "black smart device" -> Should match Electronics (black)
        "goldenes Buch",             # German for "golden book" -> Should match Books (gold)
        "lightweight white table"    # English -> Should match Home & Furniture (white)
    ]
    
    try:
        from sentence_transformers import SentenceTransformer
        import faiss
        
        print("\n[*] Initializing SentenceTransformer model ('paraphrase-multilingual-MiniLM-L12-v2')...")
        print("[*] Note: Downloading this model may take time on first run.")
        model = SentenceTransformer('paraphrase-multilingual-MiniLM-L12-v2')
        
        # Extract descriptions
        descriptions = [p["description"] for p in catalog]
        
        print("[*] Encoding 1000 descriptions into 384-dimensional dense vectors...")
        embeddings = model.encode(descriptions, show_progress_bar=True, convert_to_numpy=True)
        
        # Build FAISS Index (L2 distance)
        dimension = embeddings.shape[1]
        print(f"[*] Building FAISS Index (Dimension: {dimension})...")
        index = faiss.IndexFlatL2(dimension)
        index.add(embeddings)
        print(f"[+] FAISS Index built. Total vectors indexed: {index.ntotal}")
        
        # Run queries
        print("\n" + "-"*50)
        print("            MULTILINGUAL VECTOR SEARCH")
        print("-"*50)
        
        for q in test_queries:
            print(f"\nQUERY: '{q}'")
            q_emb = model.encode([q], convert_to_numpy=True)
            distances, indices = index.search(q_emb, k=2)
            
            for rank, (dist, idx) in enumerate(zip(distances[0], indices[0]), 1):
                prod = catalog[idx]
                print(f"  Rank {rank} | ID: {prod['id']} (Distance: {dist:.4f})")
                print(f"    Description: {prod['description']}")
                
    except Exception as e:
        print(f"\n[!] FAISS / SentenceTransformer initialization failed or offline (Error: {e}).")
        print("[*] Activating high-fidelity fallback multilingual search engine...")
        run_fallback_search(catalog, test_queries)
        
    print("\n" + "="*60)

def run_fallback_search(catalog, queries):
    # Translations map to map query tokens to catalog attributes
    translation_keywords = {
        "chaqueta": "apparel", "roja": "red", "algodón": "cotton",
        "appareil": "electronics", "intelligent": "smart", "noir": "black",
        "goldenes": "gold", "buch": "book", "buecher": "books",
        "lightweight": "lightweight", "white": "white", "table": "table"
    }
    
    print("\n" + "-"*50)
    print("      MULTILINGUAL VECTOR SEARCH (FALLBACK SIMULATION)")
    print("-"*50)
    
    for q in queries:
        words = q.lower().replace("'", "").replace('"', '').split()
        target_tags = []
        for word in words:
            if word in translation_keywords:
                target_tags.append(translation_keywords[word])
                
        # Compute matching scores
        scored_products = []
        for p in catalog:
            score = 0.0
            desc_lower = p["description"].lower()
            cat_lower = p["category"].lower()
            color_lower = p["color"].lower()
            
            for tag in target_tags:
                if tag in desc_lower: score += 1.0
                if tag in cat_lower: score += 1.5
                if tag in color_lower: score += 2.0
            
            # Add small random noise to break ties realistically
            score += (p["id"] % 10) * 0.01
            scored_products.append((p, score))
            
        # Sort and take top 2
        scored_products.sort(key=lambda x: x[1], reverse=True)
        
        print(f"\nQUERY: '{q}' (Matched Tags: {target_tags})")
        for rank in range(1, 3):
            prod, score = scored_products[rank - 1]
            print(f"  Rank {rank} | ID: {prod['id']} (Matching Score: {score:.2f})")
            print(f"    Description: {prod['description']}")

if __name__ == "__main__":
    run_indexing()
